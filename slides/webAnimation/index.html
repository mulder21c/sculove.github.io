setTimeout


function animate(el, x, y, duration) {
	var startTime = Date.now();
	var now, per;
	(function loop() {
		now = Date.now();
		if (now >= startTime + duration) {
			el.style.left = x + "px";
			el.style.top = y + "px";
			return;
		}
		per =  (now - startTime) / duration;
		el.style.left = (x * per) + "px";
		el.style.top = (y * per) + "px";
		setTimeout(loop, 16);
	})();
}
